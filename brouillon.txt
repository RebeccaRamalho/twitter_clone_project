exports.getTwenty = (callback) => {
    db.query('SELECT * FROM tweets ORDER BY id DESC LIMIT 20', (error, result) => {
        if(error){
            console.log('error: ', error);
            callback(error, null);
            return;
        }

        console.log(result);
        callback(null, result);
    })

}
